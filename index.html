Revenue YTD.dax 
-- Year-to-date revenue with context transition
Revenue YTD :=
    CALCULATE (
        [Total Revenue],
        DATESYTD ( 'Calendar'[Date] )
    )

-- Compare to same period last year
Revenue PY :=
    CALCULATE (
        [Total Revenue],
        SAMEPERIODLASTYEAR ( 'Calendar'[Date] )
    )

-- Growth percentage, safe division
YoY Growth % :=
    DIVIDE (
        [Revenue YTD] - [Revenue PY],
        [Revenue PY]
    )
